kit : !Kit{
    uri : !String
    name : !String
    description : ?String
    icon : ?String // emoji | logogram | url
    image : ?String // url
    preview : ?String // url
    version : ?String // empty is latest
    remote : ?String // url for remote fetching
    homepage : ?String // url
    license : ?String // spdx id | url
    authors : *Author[
        name : !String
        email : !String
    ]
    types : *Type[
        name : !String
        description : ?String
        icon : ?String // emoji | logogram | url
        image : ?String // url
        variant : ?String // empty is default
        stock : !Float // default is +infinity
        virtual : ?Boolean // default false
        scalable : ?Boolean // default true
        mirrorable : ?Boolean // default true
        unit : !String // e.g., mm, cm, m
        location : ?Location{
            longitude : ?Float
            latitude : ?Float
        }
        representations : +Representation[
            url : !String
            description : ?String
            tags : *String[] // default is empty list (meaning default LOD)
            attributes : *Attribute[]
        ]
        ports : +Port[
            id : !String // empty is default
            description : ?String
            mandatory : ?Boolean // default false
            family : ?String // For explicit compatibility
            compatibleFamilies : *String[] // Empty list means compatible with all
            point : !Point{
                x : !Float
                y : !Float
                z : !Float
            }
            direction : !Vector{
                x : !Float
                y : !Float
                z : !Float
            }
            t : !Float // [0,1[ for diagram ring position
            props : *Prop[
                key : !String // quality key
                value : !String // number | text
                unit : ?String
                attributes : *Attribute[]
            ]
            attributes : *Attribute[]
        ]
        authors: *AuthorId[
            email : !String
        ]
        attributes : *Attribute[]
        created : !String // date
        updated : !String // date
    ]
    qualities : *Quality[
        key : !String
        name : !String
        description : ?String
        uri : ?String
        scalable : ?Boolean // default false
        kind : !QualityKind // enum: General, Design, Type, Piece, Connection, Port
        si : ?String
        imperial : ?String
        min : ?Float
        minExcluded : ?Boolean // default true
        max : ?Float
        maxExcluded : ?Boolean // default true
        default : ?Float
        formula : ?String
        benchmarks : *Benchmark[
            name : !String
            icon : ?String
            min : ?Float
            minExcluded : ?Boolean // default false
            max : ?Float
            maxExcluded : ?Boolean // default false
        ]
        attributes : *Attribute[]
    ]
    designs : *Design[
        name : !String
        description : ?String
        icon : ?String // emoji | logogram | url
        image : ?String // url
        variant : ?String // empty is default
        view : ?String // empty is default
        unit : !String // e.g., mm, cm, m
        location : ?Location{
            longitude : ?Float
            latitude : ?Float
        }
        pieces : +Piece[
            id : !String
            description : ?String
            type : !TypeId{
                name : !String
                variant : ?String
            }
            design : ?DesignId{
                name : !String
                variant : ?String
                view : ?String
            }
            plane : ?Plane{
                origin : !Point{
                    x : !Float
                    y : !Float
                    z : !Float
                }
                xAxis : !Vector{
                    x : !Float
                    y : !Float
                    z : !Float
                }
                yAxis : !Vector{
                    x : !Float
                    y : !Float
                    z : !Float
                }
            }
            center : ?Coord{
                x : !Float
                y : !Float
            }
            scale : ?Float // default 1.0
            mirrorPlane : ?Plane{
                origin : !Point{
                    x : !Float
                    y : !Float
                    z : !Float
                }
                xAxis : !Vector{
                    x : !Float
                    y : !Float
                    z : !Float
                }
                yAxis : !Vector{
                    x : !Float
                    y : !Float
                    z : !Float
                }
            }
            hidden : ?Boolean // default false
            locked : ?Boolean // default false
            color : ?String // hex color
            attributes : *Attribute[]
        ]
        connections : +Connection[
            connected : !Side{
                piece : !PieceId{ id : !String }
                port : !PortId{ id : !String }
            }
            connecting : !Side{
                piece : !PieceId{ id : !String }
                port : !PortId{ id : !String }
            }
            description : ?String
            gap : ?Float
            shift : ?Float
            rise : ?Float
            rotation : ?Float // degrees
            turn : ?Float // degrees
            tilt : ?Float // degrees
            x : ?Float // diagram offset x
            y : ?Float // diagram offset y
            attributes : *Attribute[]
        ]
        layers : *Layer[
            name : !String
            description : ?String
            color : ?String // hex color
        ]
        groups : *Group[
            name : ?String
            description : ?String
            color : ?String // hex color
            pieces : *PieceId[
                id : !String
            ]
        ]
        authors: *AuthorId[
            email : !String
        ]
        stats : *Stat[
            key : !String // quality key
            unit : ?String
            min : ?Float
            minExcluded : ?Boolean // default false
            max : ?Float
            maxExcluded : ?Boolean // default false
        ]
        attributes : *Attribute[]
        created : !String // date
        updated : !String // date
    ]
    attributes : *Attribute[
        name : !String
        value : ?String // No value means true
        unit : ?String
        definition : ?String // text | uri
    ]
    created : !String // date
    updated : !String // date
}
