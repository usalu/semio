kit : !Kit{
    uri : !String
    name : !String
    description : ?String
    icon : ?String // emoji | logogram | url
    image : ?String // url
    preview : ?String // url
    version : ?String // empty is latest
    remote : ?String // url for remote fetching
    homepage : ?String // url
    license : ?String // spdx id | url
    created : !String // date
    updated : !String // date
    authors : *Author[
        name : !String
        email : !String
    ]
    types : *Type[
        name : !String
        description : ?String
        icon : ?String // emoji | logogram | url
        image : ?String // url
        variant : ?String // empty is default
        stock : !Float // default is +infinity
        virtual : ?Boolean // default false
        unit : !String // e.g., mm, cm, m
        created : !String // date
        updated : !String // date
        longitude : ?Float
        latitude : ?Float
        representations : +Representation[
            url : !String
            description : ?String
            tags : *String[] // default is empty list (meaning default LOD)
            attributes : *Attribute[]
        ]
        ports : +Port[
            id : !String // empty is default
            description : ?String
            mandatory : ?Boolean // default false
            family : ?String // For explicit compatibility
            compatibleFamilies : *String[] // Empty list means compatible with all
            point : !Point{
                x : !Float
                y : !Float
                z : !Float
            }
            direction : !Vector{
                x : !Float
                y : !Float
                z : !Float
            }
            t : !Float // [0,1[ for diagram ring position
            attributes : *Attribute[]
        ]
        attributes : *Attribute[]
    ]
    designs : *Design[
        name : !String
        description : ?String
        icon : ?String // emoji | logogram | url
        image : ?String // url
        variant : ?String // empty is default
        view : ?String // empty is default
        unit : !String // e.g., mm, cm, m
        created : !String // date
        updated : !String // date
        pieces : +Piece[
            id : !String
            description : ?String
            type : !TypeId{
                name : !String
                variant : ?String
            }
            design : ?DesignId{
                name : !String
                variant : ?String
                view : ?String
            }
            plane : ?Plane{
                origin : !Point{
                    x : !Float
                    y : !Float
                    z : !Float
                }
                xAxis : !Vector{
                    x : !Float
                    y : !Float
                    z : !Float
                }
                yAxis : !Vector{
                    x : !Float
                    y : !Float
                    z : !Float
                }
            }
            center : ?DiagramPoint{
                x : !Float
                y : !Float
            }
            scale : ?Float // default 1.0
            mirrorPlane : ?Plane{
                origin : !Point{
                    x : !Float
                    y : !Float
                    z : !Float
                }
                xAxis : !Vector{
                    x : !Float
                    y : !Float
                    z : !Float
                }
                yAxis : !Vector{
                    x : !Float
                    y : !Float
                    z : !Float
                }
            }
            hidden : ?Boolean // default false
            locked : ?Boolean // default false
            color : ?String // hex color
            attributes : *Attribute[]
        ]
        connections : +Connection[
            connected : !Side{
                piece : !PieceId{ id : !String }
                port : !PortId{ id : !String }
            }
            connecting : !Side{
                piece : !PieceId{ id : !String }
                port : !PortId{ id : !String }
            }
            description : ?String
            gap : ?Float
            shift : ?Float
            raise : ?Float
            rotation : ?Float // degrees
            turn : ?Float // degrees
            tilt : ?Float // degrees
            x : ?Float // diagram offset x
            y : ?Float // diagram offset y
            attributes : *Attribute[]
        ]
        layers : *Layer[
            name : !String
            description : ?String
            color : ?String // hex color
        ]
        groups : *Group[
            name : !String
            description : ?String
            color : ?String // hex color
            pieces : *PieceId[
                id : !String
            ]
        ]
        attributes : *Attribute[]
    ]
    attributes : *Attribute[
        name : !String
        value : ?String // No value means true
        unit : ?String
        definition : ?String // text | uri
    ]
}
